## Tujuan File Ini

Menjelaskan bagaimana sistem mengambil data historis nyata,
mengidentifikasi episode yang relevan,
dan menghitung probabilitas berbasis fakta.

## Prinsip Engine Ini

### Data nyata, bukan simulasi
Semua angka yang keluar dari engine ini
berasal dari data harga historis aktual
via Yahoo Finance (yfinance).

Tidak ada angka yang dikarang.
Tidak ada probabilitas yang dihitung tanpa episode nyata.

### Episode-based, bukan model-based
Engine ini tidak menggunakan machine learning
atau model prediktif.

Engine ini menggunakan pendekatan sederhana:
*"Berapa kali kondisi seperti ini terjadi di masa lalu,
dan apa yang terjadi setelahnya?"*

Ini lebih transparan, lebih mudah diverifikasi,
dan lebih bisa dipercaya oleh user awam.

### Jujur tentang keterbatasan data
Jika episode yang ditemukan kurang dari 5,
sistem tidak menampilkan probabilitas
tapi menampilkan warning bahwa sample terlalu kecil.

## Arsitektur Engine
```python
class CounterEvidenceEngine:
    
    def __init__(self, ticker: str):
        self.ticker = ticker
        self.data = self._load_data()
    
    def _load_data(self) -> pd.DataFrame:
        return yf.download(
            self.ticker, 
            start="2015-01-01",
            auto_adjust=True
        )
    
    def get_counter_evidence(self, 
                              bias_type: str,
                              **params) -> dict:
        
        if bias_type == "FOMO":
            return self._fomo_evidence(**params)
        
        elif bias_type == "LOSS_AVERSION":
            return self._loss_aversion_evidence(**params)
        
        elif bias_type == "CONFIRMATION_BIAS":
            return self._confirmation_bias_evidence()
        
        else:
            return {"error": "Bias type tidak dikenali"}
```

## Counter Evidence Per Bias

### FOMO Counter Evidence
```python
def _fomo_evidence(self, 
                    threshold: float = 0.20,
                    window: int = 5,
                    forward_days: int = 30) -> dict:
    """
    Cari semua episode di mana harga naik >threshold%
    dalam window hari, lalu hitung apa yang terjadi
    dalam forward_days hari berikutnya.
    """
    
    data = self.data.copy()
    data['return_5d'] = data['Close'].pct_change(window)
    data['forward_return'] = (
        data['Close'].shift(-forward_days) / 
        data['Close'] - 1
    )
    
    # Identifikasi episode
    episodes = data[data['return_5d'] > threshold].copy()
    
    # Filter: minimal 5 hari jarak antar episode
    # untuk menghindari double-counting
    episodes = self._filter_consecutive(episodes, min_gap=5)
    
    if len(episodes) < 3:
        return {
            "status": "insufficient_data",
            "episodes_found": len(episodes),
            "message": "Data historis tidak cukup 
                        untuk kalkulasi reliable."
        }
    
    # Hitung outcomes
    corrections = episodes[
        episodes['forward_return'] < 0
    ]['forward_return']
    
    return {
        "status": "ok",
        "ticker": self.ticker,
        "threshold": f"{threshold*100:.0f}%",
        "window_days": window,
        "forward_days": forward_days,
        "episodes_found": len(episodes),
        "corrections_count": len(corrections),
        "correction_probability": (
            len(corrections) / len(episodes)
        ),
        "avg_correction": corrections.mean(),
        "worst_correction": corrections.min(),
        "best_outcome": episodes['forward_return'].max(),
        "episode_dates": episodes.index.tolist()
    }
```

### Loss Aversion Counter Evidence
```python
def _loss_aversion_evidence(self,
                              drawdown_threshold: float = -0.10,
                              forward_days: int = 60) -> dict:
    """
    Cari semua episode downtrend (MA20 < MA50)
    dengan drawdown > threshold,
    lalu hitung berapa lama recovery terjadi.
    """
    
    data = self.data.copy()
    data['MA20'] = data['Close'].rolling(20).mean()
    data['MA50'] = data['Close'].rolling(50).mean()
    data['drawdown'] = (
        data['Close'] / 
        data['Close'].rolling(52*5).max() - 1
    )
    
    # Identifikasi episode downtrend
    downtrend_episodes = data[
        (data['MA20'] < data['MA50']) & 
        (data['drawdown'] < drawdown_threshold)
    ].copy()
    
    downtrend_episodes = self._filter_consecutive(
        downtrend_episodes, min_gap=20
    )
    
    if len(downtrend_episodes) < 3:
        return {
            "status": "insufficient_data",
            "episodes_found": len(downtrend_episodes)
        }
    
    # Hitung recovery time per episode
    recovery_days = []
    for date in downtrend_episodes.index:
        entry_price = data.loc[date, 'Close']
        future_data = data[data.index > date].head(
            forward_days * 2
        )
        
        # Cari kapan harga kembali ke entry price
        recovery = future_data[
            future_data['Close'] >= entry_price
        ]
        
        if len(recovery) > 0:
            days = (recovery.index[0] - date).days
            recovery_days.append(days)
        else:
            recovery_days.append(None)  # belum recovery
    
    recovered = [d for d in recovery_days if d is not None]
    not_recovered = [d for d in recovery_days if d is None]
    
    return {
        "status": "ok",
        "ticker": self.ticker,
        "episodes_found": len(downtrend_episodes),
        "recovered_count": len(recovered),
        "not_recovered_count": len(not_recovered),
        "avg_recovery_days": (
            sum(recovered) / len(recovered) 
            if recovered else None
        ),
        "recovery_probability": (
            len(recovered) / len(downtrend_episodes)
        ),
        "continued_down": len([
            d for d in downtrend_episodes['drawdown'] 
            if d < drawdown_threshold * 1.5
        ])
    }
```

### Confirmation Bias Counter Evidence
```python
def _confirmation_bias_evidence(self) -> dict:
    """
    Untuk confirmation bias, counter evidence
    bukan berupa episode historis —
    tapi berupa data fundamental yang
    biasanya tidak dicari user.
    """
    
    ticker_obj = yf.Ticker(self.ticker)
    info = ticker_obj.info
    
    # Data yang biasanya diabaikan saat confirmation bias
    return {
        "status": "ok",
        "ticker": self.ticker,
        "fundamental_data": {
            "pe_ratio": info.get('trailingPE', 'N/A'),
            "debt_to_equity": info.get(
                'debtToEquity', 'N/A'
            ),
            "revenue_growth": info.get(
                'revenueGrowth', 'N/A'
            ),
            "profit_margins": info.get(
                'profitMargins', 'N/A'
            ),
            "52w_high": info.get('fiftyTwoWeekHigh', 'N/A'),
            "52w_low": info.get('fiftyTwoWeekLow', 'N/A'),
            "current_price": info.get(
                'currentPrice', 'N/A'
            ),
        },
        "questions_to_ask": [
            f"Apa risiko terbesar {self.ticker} saat ini?",
            f"Siapa yang menjual {self.ticker} dan mengapa?",
            "Apa yang bisa membuat thesis ini salah?",
            f"Bagaimana performa saat IHSG turun 10%?"
        ]
    }
```

## Visualisasi Output
```python
def plot_fomo_episodes(self, evidence: dict) -> None:
    """
    Buat chart yang menunjukkan:
    1. Harga historis aset
    2. Titik-titik episode serupa
    3. Outcome 30 hari setelah setiap episode
    """
    
    import matplotlib.pyplot as plt
    import streamlit as st
    
    fig, ax = plt.subplots(figsize=(12, 6))
    
    # Plot harga historis
    ax.plot(
        self.data.index, 
        self.data['Close'],
        color='#2E86AB', 
        linewidth=1,
        alpha=0.7,
        label='Harga'
    )
    
    # Plot episode markers
    for date in evidence['episode_dates']:
        ax.axvline(
            x=date, 
            color='#E84855', 
            alpha=0.3,
            linewidth=1
        )
    
    ax.set_title(
        f"Episode Serupa: {self.ticker} "
        f"({evidence['episodes_found']} kejadian)",
        fontsize=14
    )
    ax.set_xlabel("Tanggal")
    ax.set_ylabel("Harga (IDR)")
    ax.legend()
    
    st.pyplot(fig)
```

## Helper Functions
```python
def _filter_consecutive(self, 
                          df: pd.DataFrame,
                          min_gap: int = 5
                          ) -> pd.DataFrame:
    """
    Filter episode yang terlalu berdekatan
    untuk menghindari double-counting.
    """
    
    if len(df) == 0:
        return df
    
    filtered_indices = [df.index[0]]
    
    for idx in df.index[1:]:
        last = filtered_indices[-1]
        gap = (idx - last).days
        
        if gap >= min_gap:
            filtered_indices.append(idx)
    
    return df.loc[filtered_indices]


def format_evidence_text(self, 
                          evidence: dict,
                          bias_type: str) -> str:
    """
    Format evidence dict menjadi teks
    yang siap ditampilkan di UI.
    """
    
    if evidence['status'] == 'insufficient_data':
        return (
            f"⚠️ Data historis {self.ticker} "
            f"tidak cukup untuk kalkulasi reliable.\n"
            f"Episode ditemukan: "
            f"{evidence['episodes_found']} "
            f"(minimum: 3)\n\n"
            f"Lakukan riset fundamental secara manual."
        )
    
    if bias_type == "FOMO":
        prob = evidence['correction_probability']
        avg  = evidence['avg_correction']
        
        return (
            f"Data historis {self.ticker} "
            f"({evidence['ticker']}):\n"
            f"→ Episode serupa  : "
            f"{evidence['episodes_found']} kejadian\n"
            f"→ Diikuti koreksi : "
            f"{evidence['corrections_count']} kejadian\n"
            f"→ Probabilitas    : {prob:.0%}\n"
            f"→ Rata-rata koreksi: {avg:.1%}\n"
            f"→ Koreksi terburuk : "
            f"{evidence['worst_correction']:.1%}"
        )
```

# Test case yang harus lulus sebelum MVP deploy:

def test_fomo_evidence():
    engine = CounterEvidenceEngine("BBCA.JK")
    result = engine.get_counter_evidence(
        "FOMO", 
        threshold=0.15,
        window=5,
        forward_days=30
    )
    
    assert result['status'] == 'ok'
    assert result['episodes_found'] >= 3
    assert 0 <= result['correction_probability'] <= 1
    assert result['avg_correction'] < 0
    print("✅ FOMO evidence test passed")

    # GOTO IPO 2022, data pendek

        "FOMO",
        threshold=0.30,
        window=5
    )
    
    # Harus handle dengan baik, tidak crash
    assert result['status'] in ['ok', 'insufficient_data']
    print("✅ Insufficient data test passed")
